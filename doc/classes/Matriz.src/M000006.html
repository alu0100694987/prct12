  def +(other) 
    raise TypeError, 'Las matrices no son del mismo tamanyo' unless (@rows==other.rows && @cols==other.cols)
      result=Array.new
      @rows.times do |i|
        @cols.times do |j|
          result << (self[i,j] + other[i,j])
        end
      end
      Matriz.vector(@rows, @cols, result)
  end
