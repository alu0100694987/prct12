  def initialize(rows, cols, ele)  
    super(rows, cols)
    
    @val, @f, @c = Array.new, Array.new, Array.new

    for i in 0...(@rows*@cols)
      if(ele[i]!=0)
        @val << (ele[i])
        @f << (i/@cols)
        @c << (i%@cols)
      end
    end
  end
