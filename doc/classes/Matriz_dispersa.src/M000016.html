  def ==(other)
    if (@val.length != other.length || @rows != other.rows || @cols != other.cols)
      return false
    end
    for pos in 0...@val.length
      if (@val[pos] != other(@f[pos],@c[pos]))
        return false
      end
    end
    true
  end
